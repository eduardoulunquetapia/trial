               function formatDate(userDate) {
                    var pattern = /(0[1-9]|[12]\d|3[01])\/(0[1-9]|1[0-2])\/([12]\d{3})/g,
                         value = "";
     
                    if (pattern.test(userDate)) {
                         var date = userDate.split("/")
                         value = date[2] + date[1] + date[0];
                    } else {
                         value = "error en formato formato"
                    }
                    return value;
               }
